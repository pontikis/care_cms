404 function wpautop($pee, $br = true) {
405  $pre_tags = array();
406 
407  if(trim($pee) === '')
408  return '';
409 
410  $pee = $pee . "\n"; // just to make things a little easier, pad the end
411 
412  if(strpos($pee, '<pre') !== false) {
413  $pee_parts = explode('</pre>', $pee);
414  $last_pee = array_pop($pee_parts);
415  $pee = '';
416  $i = 0;
417 
418  foreach($pee_parts as $pee_part) {
419  $start = strpos($pee_part, '<pre');
420 
421  // Malformed html?
422  if($start === false) {
423  $pee .= $pee_part;
424  continue;
425  }
426 
427  $name = "<pre wp-pre-tag-$i></pre>";
428  $pre_tags[$name] = substr($pee_part, $start) . '</pre>';
429 
430  $pee .= substr($pee_part, 0, $start) . $name;
431  $i++;
432  }
433 
434  $pee .= $last_pee;
435  }
436 
437  $pee = preg_replace('|<br />\s*<br />|', "\n\n", $pee);
438  // Space things out a little
439  $allblocks = '(?:table|thead|tfoot|caption|col|colgroup|tbody|tr|td|th|div|dl|dd|dt|ul|ol|li|pre|select|option|form|map|area|blockquote|address|math|style|p|h[1-6]|hr|fieldset|noscript|legend|section|article|aside|hgroup|header|footer|nav|figure|figcaption|details|menu|summary)';
440  $pee = preg_replace('!(<' . $allblocks . '[^>]*>)!', "\n$1", $pee);
441  $pee = preg_replace('!(</' . $allblocks . '>)!', "$1\n\n", $pee);
442  $pee = str_replace(array("\r\n", "\r"), "\n", $pee); // cross-platform newlines
443  if(strpos($pee, '<object') !== false) {
444  $pee = preg_replace('|\s*<param([^>]*)>\s*|', "<param$1>", $pee); // no pee inside object/embed
445  $pee = preg_replace('|\s*</embed>\s*|', '</embed>', $pee);
446  }
447  $pee = preg_replace("/\n\n+/", "\n\n", $pee); // take care of duplicates
448  // make paragraphs, including one at the end
449  $pees = preg_split('/\n\s*\n/', $pee, -1, PREG_SPLIT_NO_EMPTY);
450  $pee = '';
451  foreach($pees as $tinkle)
452  $pee .= '<p>' . trim($tinkle, "\n") . "</p>\n";
453  $pee = preg_replace('|<p>\s*</p>|', '', $pee); // under certain strange conditions it could create a P of entirely whitespace
454  $pee = preg_replace('!<p>([^<]+)</(div|address|form)>!', "<p>$1</p></$2>", $pee);
455  $pee = preg_replace('!<p>\s*(</?' . $allblocks . '[^>]*>)\s*</p>!', "$1", $pee); // don't pee all over a tag
456  $pee = preg_replace("|<p>(<li.+?)</p>|", "$1", $pee); // problem with nested lists
457  $pee = preg_replace('|<p><blockquote([^>]*)>|i', "<blockquote$1><p>", $pee);
458  $pee = str_replace('</blockquote></p>', '</p></blockquote>', $pee);
459  $pee = preg_replace('!<p>\s*(</?' . $allblocks . '[^>]*>)!', "$1", $pee);
460  $pee = preg_replace('!(</?' . $allblocks . '[^>]*>)\s*</p>!', "$1", $pee);
461  if($br) {
462  $pee = preg_replace_callback('/<(script|style).*?<\/\\1>/s', '_autop_newline_preservation_helper', $pee);
463  $pee = preg_replace('|(?<!<br />)\s*\n|', "<br />\n", $pee); // optionally make line breaks
464  $pee = str_replace('<WPPreserveNewline />', "\n", $pee);
465  }
466  $pee = preg_replace('!(</?' . $allblocks . '[^>]*>)\s*<br />!', "$1", $pee);
467  $pee = preg_replace('!<br />(\s*</?(?:p|li|div|dl|dd|dt|th|pre|td|ul|ol)[^>]*>)!', '$1', $pee);
468  $pee = preg_replace("|\n</p>$|", '</p>', $pee);
469 
470  if(!empty($pre_tags))
471  $pee = str_replace(array_keys($pre_tags), array_values($pre_tags), $pee);
472 
473  return $pee;
474 }
