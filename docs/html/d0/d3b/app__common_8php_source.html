603

637  if(strpos($pee, '<object') !== false) {
638  $pee = preg_replace('|\s*<param([^>]*)>\s*|', "<param$1>", $pee); // no pee inside object/embed
639  $pee = preg_replace('|\s*</embed>\s*|', '</embed>', $pee);
640  }
641  $pee = preg_replace("/\n\n+/", "\n\n", $pee); // take care of duplicates
642  // make paragraphs, including one at the end
643  $pees = preg_split('/\n\s*\n/', $pee, -1, PREG_SPLIT_NO_EMPTY);
644  $pee = '';
645  foreach($pees as $tinkle)
646  $pee .= '<p>' . trim($tinkle, "\n") . "</p>\n";
647  $pee = preg_replace('|<p>\s*</p>|', '', $pee); // under certain strange conditions it could create a P of entirely whitespace
648  $pee = preg_replace('!<p>([^<]+)</(div|address|form)>!', "<p>$1</p></$2>", $pee);
649  $pee = preg_replace('!<p>\s*(</?' . $allblocks . '[^>]*>)\s*</p>!', "$1", $pee); // don't pee all over a tag
650  $pee = preg_replace("|<p>(<li.+?)</p>|", "$1", $pee); // problem with nested lists
651  $pee = preg_replace('|<p><blockquote([^>]*)>|i', "<blockquote$1><p>", $pee);
652  $pee = str_replace('</blockquote></p>', '</p></blockquote>', $pee);
653  $pee = preg_replace('!<p>\s*(</?' . $allblocks . '[^>]*>)!', "$1", $pee);
654  $pee = preg_replace('!(</?' . $allblocks . '[^>]*>)\s*</p>!', "$1", $pee);
655  if($br) {
656  $pee = preg_replace_callback('/<(script|style).*?<\/\\1>/s', array($this, '_autop_newline_preservation_helper'), $pee);
657  $pee = preg_replace('|(?<!<br />)\s*\n|', "<br />\n", $pee); // optionally make line breaks
658  $pee = str_replace('<WPPreserveNewline />', "\n", $pee);
659  }
660  $pee = preg_replace('!(</?' . $allblocks . '[^>]*>)\s*<br />!', "$1", $pee);
661  $pee = preg_replace('!<br />(\s*</?(?:p|li|div|dl|dd|dt|th|pre|td|ul|ol)[^>]*>)!', '$1', $pee);
662  $pee = preg_replace("|\n</p>$|", '</p>', $pee);
663 
664  if(!empty($pre_tags))
665  $pee = str_replace(array_keys($pre_tags), array_values($pre_tags), $pee);
666 
667  return $pee;
